гостиница (hotel) (
	айди: serial
	название: varchar(255)
	описание: text
	список фотографий (упорядоченный): hotel_photo_bind
	населённый пункт: locality
	yлица: varchar(255)
	контактная информация: varchar(255)
	поддерживаемые способы оплаты: hotel_payment_type_bind
	расстояние до центра населённого пункта в километрах: unsigned int
)

гостиничный номер (room) (
	айди: serial
	тип номера: room_type
)

тип гостиничного номера (room_type) (
	айди: serial
	гостиница: hotel
	название: varchar(255)
	цена за сутки в рублях: int unsigned
	типы кроватей: room_type_bed_bind
	удобства: room_type_facilities_bind
	УНИКАЛЬНЫЕ ПОЛЯ: гостинца и название (можно сделать эту пару как PK)
)

статус гостиничного номера на конкретную дату (room_status) (
	гостиничный номер: room
	дата: date
	статус: tinyint unsigned (свободен, занят, на ремонте и т. д.)
	PK: гостиничный номер и дата
)

скидка для типа гостиничного номера на конкретную дату (room_type_discount) (
	тип гостиничного номера: room_type
	дата: date
	процент скидки: tinyint unsigned
	PK: тип гостиничного номера и дата
)

удобства (facilities) (
	айди: serial
	название: varchar(255)
)

тип кровати (bed) (
	айди: serial
	количество спальных мест: tinyint unsigned
)

фото (photo) (
	айди: serial
	путь до файла: varchar(255)
)

населённый пункт (locality) (
	айди: serial
	название: varchar(255)
)

заказ (order) (
	айди: serial
	пользователь: user
	гостиничный номер: room
	дата заезда: date
	дата выезда: date
	оплата: payment
)

пользователь (user) (
	логин: varchar(255)
	хэшированный пароль: varchar(255)
	имя: varchar(255)
	фамилия: varchar(255)
	номер телефона: varchar(255)
	почта: varchar(255)
)

оплата (payment) (
	айди: serial
	статус оплаты: tinyint unsigned (оплачен, оплачен частично, забронирован)
	суммарная цена в рублях: unsigned int
	сколько уже оплачено рублей: unsigned int
	сколько ещё необходимо заплатить рублей: unsigned int
)

способ оплаты (payment_type) (
	айди: serial
	название: varchar(255)
)
